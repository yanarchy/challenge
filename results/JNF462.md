function digitalSum(n)
{
	if(n===0){
		return 0;
	}
	var result=n%10
	var result1=Math.floor(n/10);
	return result+digitalSum(result1);
}

function fib(n){
	if(n===0 ){
		return 0;
	}
	if(n===1){
		return 1;
	}
	return fib(n-1)+fib(n-2);
}
